This function returns a list of nearby BluPOINT devices.

Sample Code

mBluIDSDK_Client.let { sdkClient ->
  sdkClient.discoverDevices(
     BluPOINTFilter(
           -80,
           1000
     )
 ) { bleDevices -> Unit
     for (device in bleDevices) {
         print(device)
         }
     }
}
